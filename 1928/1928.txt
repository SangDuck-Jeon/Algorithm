void Decoding (char a, char b, char c, char d)
{
    char shift[4];
    shift[0] = ((a<<2)&0xFC) + ((b>>4)&0x03);
    shift[1] = ((b<<4)&0xF0) + ((c>>2)&0x0F);
    shift[2] = ((c<<6)&0xC0) + (d&0x3F);
    printf("%c%c%c", shift[0], shift[1], shift[2]);
}
